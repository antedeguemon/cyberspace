25 Lists in Elixir are effectively linked lists, which means
26 they are internally represented in pairs containing the
27 head and the tail of a list:
28
29 iex> [head | tail] = [1, 2, 3]
30 iex> head
31 1
32 iex> tail
33 [2, 3]
34
35 Similarly, we could write the list `[1, 2, 3]` using only
36 such pairs (called cons cells):
37
38 iex> [1 | [2 | [3 | []]]]
39 [1, 2, 3]
40
41 Some lists, called improper lists, do not have an empty list as
42 the second element in the last cons cell:
43
44 iex> [1 | [2 | [3 | 4]]]
45 [1, 2, 3 | 4]
46
47 Although improper lists are generally avoided, they are used in some
48 special circumstances like iodata and chardata entities (see the `IO` module).
49
50 Due to their cons cell based representation, prepending an element
51 to a list is always fast (constant time), while appending becomes
52 slower as the list grows in size (linear time):
53
54 iex> list = [1, 2, 3]
55 iex> [0 | list] # fast
56 [0, 1, 2, 3]
57 iex> list ++ [4] # slow
58 [1, 2, 3, 4]
59
60 Most of the functions in this module work in linear time. This means that,
61 that the time it takes to perform an operation grows at the same rate as the
62 length of the list. For example `length/1` and `last/1` will run in linear
63 time because they need to iterate through every element of the list, but
64 `first/1` will run in constant time because it only needs the first element.
65
66 Lists also implement the `Enumerable` protocol, so many functions to work with
67 lists are found in the `Enum` module. Additionally, the following functions and
68 operators for lists are found in `Kernel`:
69
70 * `++/2`
71 * `--/2`
72 * `hd/1`
73 * `tl/1`
74 * `in/2`

77 ## Charlists
78
79 If a list is made of non-negative integers, where each integer represents a
80 Unicode code point, the list can also be called a charlist. These integers
81 must:
82
83 * be within the range `0..0x10FFFF` (`0..1_114_111`);
84 * and be out of the range `0xD800..0xDFFF` (`55_296..57_343`), which is
85 reserved in Unicode for UTF-16 surrogate pairs.
86
87 Elixir uses single quotes to define charlists:
88
89 iex> 'héllo'
90 [104, 233, 108, 108, 111]
91
92 In particular, charlists will be printed back by default in single
93 quotes if they contain only printable ASCII characters:
94
95 iex> 'abc'
96 'abc'
97
98 Even though the representation changed, the raw data does remain a list of
99 numbers, which can be handled as such:
100
101 iex> inspect('abc', charlists: :as_list)
102 "[97, 98, 99]"
103 iex> Enum.map('abc', fn num -> 1000 + num end)
104 [1097, 1098, 1099]
105
106 You can use the `IEx.Helpers.i/1` helper to get a condensed rundown on
107 charlists in IEx when you encounter them, which shows you the type, description
108 and also the raw representation in one single summary.
109
110 The rationale behind this behaviour is to better support
111 Erlang libraries which may return text as charlists
112 instead of Elixir strings. In Erlang, charlists are the default
113 way of handling strings, while in Elixir it's binaries. One
114 example of such functions is `Application.loaded_applications/0`:
115
116 Application.loaded_applications()
117 #=> [
118 #=> {:stdlib, 'ERTS CXC 138 10', '2.6'},
119 #=> {:compiler, 'ERTS CXC 138 10', '6.0.1'},

121 #=> {:kernel, 'ERTS CXC 138 10', '4.1'},
122 #=> {:logger, 'logger', '1.0.0'}

1056 @doc """
1057 Returns a keyword list that represents an *edit script*.
1058
1059 The algorithm is outlined in the
1060 "An O(ND) Difference Algorithm and Its Variations" paper by E. Myers.
1061
1062 An *edit script* is a keyword list. Each key describes the "editing action" to
1063 take in order to bring `list1` closer to being equal to `list2`; a key can be
1064 `:eq`, `:ins`, or `:del`. Each value is a sublist of either `list1` or `list2`
1065 that should be inserted (if the corresponding key `:ins`), deleted (if the
1066 corresponding key is `:del`), or left alone (if the corresponding key is
1067 `:eq`) in `list1` in order to be closer to `list2`.
1068
1069 See `myers_difference/3` if you want to handle nesting in the diff scripts.
1070
1071 ## Examples
1072
1073 iex> List.myers_difference([1, 4, 2, 3], [1, 2, 3, 4])
1074 [eq: [1], del: [4], eq: [2, 3], ins: [4]]
1075
1076 """
1077 @doc since: "1.4.0"
1078 @spec myers_difference(list, list) :: [{:eq | :ins | :del, list}]
1079 def myers_difference(list1, list2) when is_list(list1) and is_list(list2) do
1080 myers_difference_with_diff_script(list1, list2, nil)
1081 end
1082
1083 @doc """
1084 Returns a keyword list that represents an *edit script* with nested diffs.
1085
1086 This is an extension of `myers_difference/2` where a `diff_script` function
1087 can be given in case it is desired to compute nested differences. The function
1088 may return a list with the inner edit script or `nil` in case there is no
1089 such script. The returned inner edit script will be under the `:diff` key.
1090
1091 ## Examples
1092
1093 iex> List.myers_difference(["a", "db", "c"], ["a", "bc"], &String.myers_difference/2)
1094 [eq: ["a"], diff: [del: "d", eq: "b", ins: "c"], del: ["c"]]
1095
1096 """
1097 @doc since: "1.8.0"
1098 @spec myers_difference(list, list, (term, term -> script | nil)) :: script
1099 when script: [{:eq | :ins | :del | :diff, list}]
1100 def myers_difference(list1, list2, diff_script)
1101 when is_list(list1) and is_list(list2) and is_function(diff_script) do
1102 myers_difference_with_diff_script(list1, list2, diff_script)
1103 end
1104
1105 defp myers_difference_with_diff_script(list1, list2, diff_script) do
1106 path = {0, list1, list2, []}
1107 find_script(0, length(list1) + length(list2), [path], diff_script)
1108 end
1109
1110 defp find_script(envelope, max, paths, diff_script) do
1111 case each_diagonal(-envelope, envelope, paths, [], diff_script) do
1112 {:done, edits} -> compact_reverse(edits, [])
1113 {:next, paths} -> find_script(envelope + 1, max, paths, diff_script)
1114 end
1115 end
1116
1117 defp compact_reverse([], acc), do: acc
1118
1119 defp compact_reverse([{:diff, _} = fragment | rest], acc) do
1120 compact_reverse(rest, [fragment | acc])
1121 end
1122
1123 defp compact_reverse([{kind, elem} | rest], [{kind, result} | acc]) do
1124 compact_reverse(rest, [{kind, [elem | result]} | acc])
1125 end
1126
1127 defp compact_reverse(rest, [{:eq, elem}, {:ins, elem}, {:eq, other} | acc]) do
1128 compact_reverse(rest, [{:ins, elem}, {:eq, elem ++ other} | acc])
1129 end
1130
1131 defp compact_reverse([{kind, elem} | rest], acc) do
1132 compact_reverse(rest, [{kind, [elem]} | acc])
1133 end
1134
1135 defp each_diagonal(diag, limit, _paths, next_paths, _diff_script) when diag > limit do
1136 {:next, :lists.reverse(next_paths)}
1137 end
1138
1139 defp each_diagonal(diag, limit, paths, next_paths, diff_script) do
1140 {path, rest} = proceed_path(diag, limit, paths, diff_script)
1141
1142 case follow_snake(path) do
1143 {:cont, path} -> each_diagonal(diag + 2, limit, rest, [path | next_paths], diff_script)
1144 {:done, edits} -> {:done, edits}
1145 end
1146 end
1147
1148 defp proceed_path(0, 0, [path], _diff_script), do: {path, []}
1149
1150 defp proceed_path(diag, limit, [path | _] = paths, diff_script) when diag == -limit do
1151 {move_down(path, diff_script), paths}
1152 end
1153
1154 defp proceed_path(diag, limit, [path], diff_script) when diag == limit do
1155 {move_right(path, diff_script), []}
1156 end
1157
1158 defp proceed_path(_diag, _limit, [path1, path2 | rest], diff_script) do
1159 if elem(path1, 0) > elem(path2, 0) do
1160 {move_right(path1, diff_script), [path2 | rest]}
1161 else
1162 {move_down(path2, diff_script), [path2 | rest]}
1163 end
1164 end
1165
1166 defp move_right({y, [elem1 | rest1] = list1, [elem2 | rest2], edits}, diff_script)
1167 when diff_script != nil do
1168 if diff = diff_script.(elem1, elem2) do
1169 {y + 1, rest1, rest2, [{:diff, diff} | edits]}
1170 else
1171 {y, list1, rest2, [{:ins, elem2} | edits]}
1172 end
1173 end
1174
1175 defp move_right({y, list1, [elem | rest], edits}, _diff_script) do
1176 {y, list1, rest, [{:ins, elem} | edits]}
1177 end
1178
1179 defp move_right({y, list1, [], edits}, _diff_script) do
1180 {y, list1, [], edits}
1181 end
1182
1183 defp move_down({y, [elem1 | rest1], [elem2 | rest2] = list2, edits}, diff_script)
1184 when diff_script != nil do
1185 if diff = diff_script.(elem1, elem2) do
1186 {y + 1, rest1, rest2, [{:diff, diff} | edits]}
1187 else
1188 {y + 1, rest1, list2, [{:del, elem1} | edits]}
1189 end
1190 end
1191
1192 defp move_down({y, [elem | rest], list2, edits}, _diff_script) do
1193 {y + 1, rest, list2, [{:del, elem} | edits]}
1194 end
1195
1196 defp move_down({y, [], list2, edits}, _diff_script) do
1197 {y + 1, [], list2, edits}
1198 end

1201 follow_snake({y + 1, rest1, rest2, [{:eq, elem} | edits]})
1202 end

1204 defp follow_snake({_y, [], [], edits}) do
